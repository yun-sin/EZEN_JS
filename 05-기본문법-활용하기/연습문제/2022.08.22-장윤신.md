## 문제 3

```js
var sum = 0;

for (let i = 1; i < 20; i++) {
  if (i % 2 == 0 || i % 3 == 0) {
    sum += i;
    console.log("i=%d 총합=%d", i, sum);
  }
}
```

## 문제 4

```js
var cases = 0;

for (let i = 1; i <= 6; i++) {
  for (let j = 1; j <= 6; j++) {
    if (i + j == 6) {
      console.log("[ %d, %d ]", i, j)

      cases++;
    }
  }
}

console.log("경우의 수는 %d개 입니다", cases)
```
